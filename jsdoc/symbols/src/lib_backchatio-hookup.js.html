 11 var RECONNECT_SCHEDULE = {min: 1, max: 300};
 12 var BUFFER_PATH = './logs/buffer.log';
 13 var EVENT_NAMES = {
 14   connected: "open",
 15   receive: "message",
 16   disconnected: "close"

 19 DISCONNECTED=0
 20 DISCONNECTING=1
 21 RECONNECTING=2
 22 CONNECTING=3
 23 CONNECTED=4

 25 /**
 26  * Creates a new hookup client 
 27  * @class
 28  * The Hookup client provides a client for the hookup server, it doesn't lock you in to using a specific message format.
 29  * The default implementation uses websockets to communicate with the server.
 30  * You can opt-in or opt-out of every feature in the hookup client through configuration.
 31  * 
 32  * Usage of the simple hookup client:
 33  *
 34  * <pre>
 35  *   var myHookup = new Hookup({uri: "ws://localhost:8080/thesocket"});
 36  *   myHookup.on("open", function() {
 37  *     console.log("The websocket to"+myHookup.uri+" is connected.")''
 38  *   }); 
 39  *   myHookup.on("close", function() {
 40  *     console.log("The websocket to"+myHookup.uri+" disconnected.")''
 41  *   }); 
 42  *   myHookup.on("data", function(data) {
 43  *     if (data.type == "text") {
 44  *       console.log("RECV: "+data.content);
 45  *       myHookup.send("ECHO: "+data.content);
 46  *     }
 47  *   });
 48  * </pre>
 49  *
 50  * @param {Object} options The options to configure this client with.
 51  */
 52 var HookupClient = function (options) {
 53   events.EventEmitter.call(this);
 54   if (typeof options === 'string') {
 55     options = { uri: options }
 56   }
 57   if (!options.uri || options.uri.trim().length == 0) throw new Error('`uri` option is required.');
 58   this.uri = options.uri;
 59   this._uri = Uri.parse(this.uri.replace(/^http/i, 'ws'));
 60   if (!this._uri.host || !((this._uri.protocol||"").match(/^ws/))) throw new Error("Invalid uri supplied.")
 61   delete options['uri'];
 62   
 63   this.reconnectSchedule = options.reconnectSchedule || RECONNECT_SCHEDULE;
 64 
 65   this._quiet = options.quiet;
 66   this._state = DISCONNECTED;
 67   this._handlers = options.handlers || [];
 68   this._wireFormat = options.wireFormat || new WireFormat();
 69   this._expectedAcks = {};
 70   this._ackCounter = 0;
 71   // this option `raiseAckEvents` is only useful in tests for the acking itself.
 72   // it raises an event when an ack is received or an ack request is prepared
 73   // it serves no other purpose than that, ack_failed events are raised independently from this option.
 74   this._raiseAckEvents = options.raiseAckEvents === true; 
 75 
 76   if (options.buffer || options.buffered) {
 77     this._buffer = options.buffer || new FileBuffer(options.bufferPath||BUFFER_PATH);
 78     this._buffer.on('data', this.send);
 79   }
 80 }

 82 util.inherits(HookupClient, events.EventEmitter);

 84 /**
 85  * The default reconnection schedule
 86  * This is an array of 2 numbers [min, max]
 87  *
 88  * @returns {Number[]} The minimum and maximum wait for a reconnection attempt
 89  */
 90 HookupClient.RECONNECT_SCHEDULE = RECONNECT_SCHEDULE
 91 /**
 92  * The default path for the file buffer to write to.
 93  *
 94  * @returns {String} The path to the file.
 95  */
 96 HookupClient.BUFFER_PATH = BUFFER_PATH

 98 _.extend(HookupClient.prototype, /** @lends HookupClient.prototype */ {
 99   /**
100    * Connect to the server
101    */
102   connect: function() {
103     if (this.state != CONNECTING && !this.isConnected()) this._establishConnection();
104   },
105   /**
106    * Disconnect from the socket, perform closing handshake if necessary
107    */
108   close: function() {
109     this._skipReconnect = true;
110     this._client.close();
111   },
112   /**
113    * Send a message over the current connection, buffer the message if not connected and the client is 
114    * configured to buffer messages.
115    *
116    * @param {String|Object} message The message to send
117    */
118   send: function(message) {
119     var m = this._prepareForSend(message);
120     if (this.isConnected()) {
121       this._client.send(m);
122     } else if(this.isBuffered()) 
123       this._buffer.write(m);
124   },
125   /**
126    * Send a message over the current connection and request that this message will be acked upon receipt by the server.
127    * Buffers the message if not connected and the client is configured to buffer messages.
128    *
129    * @param {String|Object} message The message to send
130    */
131   sendAcked: function(message, options) {
132     var timeout = (options||{})['timeout']||5000;
133     this.send({type: "needs_ack", content: message, timeout: timeout});
134   },
135   /**
136    * A flag indicating connection status of the client.
137    *
138    * @returns true when the client is connected.
139    */
140   isConnected: function() {
141     return this._state == CONNECTED;
142   },
143   /**
144    * A flag indicating if this client should fallback to buffering upon disconnection.
145    */
146   isBuffered: function() {
147     return !!this._buffer;
148   },
149   _establishConnection: function() {
150     if (this._scheduledReconnect) {
151       clearTimeout(this._scheduledReconnect);
152       this._scheduledReconnect = null;
153     }
154     if (!this.isConnected()) {
155       this._client = client = new WebSocket.Client(this.uri);
156       var self = this;
157 
158       client.onopen = function(evt) {
159         console.info("connected to " + self.uri);
160         self._connected();
161       }
162       client.onclose = function(evt) {
163         self._state = self._skipReconnect ? DISCONNECTING : RECONNECTING
164         self.emit(self._state == DISCONNECTING ? "disconnected" : "reconnecting");
165         self._reconnect();
166       }
167       client.onerror = function(evt) {
168         if (!self.quiet) console.error("Couldn't connect to " + self.uri);
169         self.emit("error", evt.data);
170       }
171       client.onmessage = function(evt) {
172         var inMsg = self._preprocessInMessage(evt.data);
173         if (inMsg)
174           self.emit("data", inMsg);
175       }
176     }
177   },
178   _reconnect: function () {
179     if (this._state == DISCONNECTING) {
180       this._doDisconnect();
181     } else {
182       this._notifiedOfReconnect = true;
183       if (this._reconnectIn && this._reconnectIn > 0 && this._reconnectIn < this.reconnectSchedule.max) {
184         var curr = this._reconnectIn;
185         var max = this.reconnectSchedule.max;
186         var next = curr < max ? curr : max;
187         if (this.reconnectSchedule.maxRetries && this.reconnectSchedule.maxRetries > 0) {
188           if (this._reconnectsLeft <= 0)
189             this.emit('error', new Error("Exhausted the retry schedule. The server at "+this.uri+" is just not there."));
190           else
191             --this._reconnectsLeft;
192         } 
193         this._doReconnect(next);
194       } else {
195         this._doDisconnect();    
196       }
197     }
198   },
199   _doDisconnect: function() {
200     this._state = DISCONNECTED;
201     this.emit('close');
202     if (this._buffer) this.buffer.close();
203   },
204   _doReconnect: function (retryIn) {
205     if (!this._scheduledReconnect) {
206       var secs = "second" + (retryIn == 1 ? "" : "s");
207       var out = retryIn < 1 ? retryIn * 1000 : retryIn;
208       console.info("connection lost, reconnecting in "+out+" "+(retryIn < 1 ? "millis" : secs));
209       var self = this;
210       this._scheduledReconnect = setTimeout(function() { self._establishConnection() }, retryIn * 1000);
211       this._reconnectIn = this._reconnectIn * 2;
212     }
213   },
214   _connected: function() {
215     if (this._buffer) this._buffer.drain();
216     this._reconnectIn = this.reconnectSchedule.min;
217     this._reconnectsLeft = 0;
218     this._notifiedOfReconnect = false;
219     this._skipReconnect = false;
220     this._state = CONNECTED;
221     this.emit("open");
222   },
223   _preprocessInMessage: function(msg) {
224     var message = this._wireFormat.parseMessage(msg);
225     if (message.type === "ack_request") {
226       this.send({type: "ack", id: message.id});
227     }
228     if (message.type === "ack") {
229       var timeout = this._expectedAcks[message.id]; 
230       if (timeout) clearTimeout(timeout);
231       delete this._expectedAcks[message.id];
232       if (this._raiseAckEvents) this.emit("ack", message);
233       return null;
234     }
235     return this._wireFormat.unwrapContent(message);
236   },
237   _prepareForSend: function(message) {
238     var out = this._wireFormat.renderMessage(message);
239     var self = this;
240     if (message.type === "needs_ack") {
241       var ackReq = {
242         content: this._wireFormat.buildMessage(message.content),
243         type: "ack_request",
244         id: ++this._ackCounter
245       };
246       this._expectedAcks[ackReq.id] = setTimeout(function (){ self.emit('ack_failed', message.content) }, message.timeout);
247       out = this._wireFormat.renderMessage(ackReq);
248       if (this._raiseAckEvents) this.emit("ack_request", ackReq);
249     }
250     return out;
251   }
252 });
